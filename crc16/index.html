Usage

use crc16::*;

// In one pass
assert_eq!(State::<ARC>::calculate(b"123456789"), 0xBB3D);

// Incrementally
let mut state = State::<ARC>::new();
state.update(b"12345");
state.update(b"6789");
assert_eq!(state.get(), 0xBB3D);
